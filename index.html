Example 1 - demonstrates simple stock keeping units of two item types, no discounts

Code:


        var sku, order, billing;

        sku     = checkout.stockKeepingUnits;
        order   = checkout.order;
        bill    = checkout.billing;

        order.clear();

        sku.set({
            A: {label: "Label for A", price: 50},
            B: {label: "Label for B", price: 30}
        });

        order.setUnits(sku.get());
        order.setRules(rules);
        order.add("B");
        order.add("A");
        order.add("B");
        order.add("A");
        order.add("A");
        order.add("A");

        bill.addCharge("order", order.charge());
        bill.update();

        $("#example1 .bill").html(bill.get());

Example 4 - demonstrates calculation of bill from orders with discount options in stock keeping unit prices

Code:


      var sku, order, billing;

      sku     = checkout.stockKeepingUnits;
      order   = checkout.order;
      billing = checkout.billing;

      order.clear();

      sku.set({
        "A": {
          label: "Label for A",
          price: 50,
          discount: {
            price: function () {
              return 130 / 3;
            },
            limit: 3
          }
        },
        "B": {
          label: "Label for B",
          price: 30,
          discount: {
            price: function () {
              return 45 / 2;
            },
            limit: 2
          }
        },
        "C": {
          label: "Label for C",
          price: 20
        },
        "D": {
          label: "Label for D",
          price: 15
        }
      });

      order.add("A").add("B").add("A").add("A").add("B").add("B").add("A").add("D").add("C").add("D").add("C");
      $("#example4 .bill").html(billing.get());
